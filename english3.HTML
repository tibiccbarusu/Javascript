1. 基本的な文法と構文の理解

データ型と変数

public class DataTypes {
    public static void main(String[] args) {
        // プリミティブ型
        int num = 10;                // 整数型
        double price = 19.99;        // 小数型
        boolean isAvailable = true;  // 真偽型

        // 参照型
        String message = "Hello, World!";  // 文字列型

        // 変数の出力
        System.out.println("Number: " + num);
        System.out.println("Price: " + price);
        System.out.println("Available: " + isAvailable);
        System.out.println("Message: " + message);
    }
}

制御構造

public class ControlStructures {
    public static void main(String[] args) {
        int score = 85;

        // if文
        if (score >= 90) {
            System.out.println("Excellent");
        } else if (score >= 75) {
            System.out.println("Good");
        } else {
            System.out.println("Needs Improvement");
        }

        // switch文
        char grade = 'B';
        switch (grade) {
            case 'A':
                System.out.println("Excellent");
                break;
            case 'B':
                System.out.println("Good");
                break;
            case 'C':
                System.out.println("Average");
                break;
            default:
                System.out.println("Fail");
        }

        // forループ
        for (int i = 1; i <= 5; i++) {
            System.out.println("Count: " + i);
        }

        // whileループ
        int count = 5;
        while (count > 0) {
            System.out.println("Countdown: " + count);
            count--;
        }
    }
}

2. オブジェクト指向プログラミング（OOP）の概念

クラスとオブジェクト

public class Dog {
    // フィールド（属性）
    String name;
    int age;

    // コンストラクタ
    public Dog(String name, int age) {
        this.name = name;
        this.age = age;
    }

    // メソッド（動作）
    public void bark() {
        System.out.println(name + " is barking!");
    }

    public static void main(String[] args) {
        // オブジェクトの生成
        Dog myDog = new Dog("Buddy", 3);
        myDog.bark();  // メソッドの呼び出し
    }
}

メソッドとインスタンスメソッド

public class MethodExample {
    // クラスメソッド
    public static void staticMethod() {
        System.out.println("This is a static method.");
    }

    // インスタンスメソッド
    public void instanceMethod() {
        System.out.println("This is an instance method.");
    }

    public static void main(String[] args) {
        // クラスメソッドの呼び出し
        MethodExample.staticMethod();

        // インスタンスメソッドの呼び出し
        MethodExample example = new MethodExample();
        example.instanceMethod();
    }
}

インスタンスの生成（通常バージョンと配列を使用したバージョン）

public class Dog {
    String name;

    // コンストラクタ
    public Dog(String name) {
        this.name = name;
    }

    public void bark() {
        System.out.println(name + " is barking!");
    }

    public static void main(String[] args) {
        // 通常のインスタンス生成
        Dog myDog = new Dog("Buddy");
        myDog.bark();

        // 配列を使用したインスタンス生成
        Dog[] dogArray = new Dog[2];
        dogArray[0] = new Dog("Max");
        dogArray[1] = new Dog("Bella");

        for (Dog dog : dogArray) {
            dog.bark();
        }
    }
}

継承、抽象クラス、インターフェース

// 抽象クラス
abstract class Animal {
    String name;

    // 抽象メソッド
    abstract void makeSound();

    public void eat() {
        System.out.println(name + " is eating.");
    }
}

// インターフェース
interface Pet {
    void play();
}

// サブクラス
class Cat extends Animal implements Pet {
    public Cat(String name) {
        this.name = name;
    }

    @Override
    void makeSound() {
        System.out.println(name + " says meow!");
    }

    @Override
    public void play() {
        System.out.println(name + " is playing.");
    }

    public static void main(String[] args) {
        Cat myCat = new Cat("Whiskers");
        myCat.makeSound();
        myCat.eat();
        myCat.play();
    }
}

3. 標準ライブラリの利用

Scannerクラスの利用

import java.util.Scanner;

public class ScannerExample {
    public static void main(String[] args) {
        // Scannerオブジェクトの作成（標準入力を読み取る）
        Scanner scanner = new Scanner(System.in);

        // 整数の入力
        System.out.print("Enter an integer: ");
        int number = scanner.nextInt();

        // 文字列の入力
        System.out.print("Enter a string: ");
        String text = scanner.next();

        // スキャナの終了
        scanner.close();

        // 入力結果の表示
        System.out.println("You entered integer: " + number);
        System.out.println("You entered string: " + text);
    }
}

Randomクラスの利用

import java.util.Random;

public class RandomExample {
    public static void main(String[] args) {
        // Randomオブジェクトの作成
        Random random = new Random();

        // 0から99までのランダムな整数を生成
        int randomNumber = random.nextInt(100);

        // 0.0から1.0までのランダムな小数を生成
        double randomDouble = random.nextDouble();

        // trueまたはfalseのランダムな値を生成
        boolean randomBoolean = random.nextBoolean();

        // 生成されたランダムな値の表示
        System.out.println("Random integer (0-99): " + randomNumber);
        System.out.println("Random double (0.0-1.0): " + randomDouble);
        System.out.println("Random boolean: " + randomBoolean);
    }
}

4. エラーハンドリング

例外処理の基本

public class ExceptionHandling {
    public static void main(String[] args) {
        try {
            int result = divide(10, 0);
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Error: Division by zero is not allowed.");
        } finally {
            System.out.println("Execution completed.");
        }
    }

    public static int divide(int a, int b) {
        return a / b;
    }
}

例外の種類

検査例外の例

import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class CheckedExceptionExample {

カスタム例外の作成

// カスタム例外クラス
class CustomException extends Exception {
    public CustomException(String message) {
        super(message);
    }
}

public class CustomExceptionExample {
    public static void main(String[] args) {
        try {
            validateAge(15);
        } catch (CustomException e) {
            System.out.println("Caught custom exception: " + e.getMessage());
        }
    }

    // 年齢を検証するメソッド
    public static void validateAge(int age) throws CustomException {
        if (age < 18) {
            throw new CustomException("Age must be 18 or older.");
        }
        System.out.println("Age is valid.");
    }
}

例外チェーン

public class ExceptionChainingExample {
    public static void main(String[] args) {
        try {
            method1();
        } catch (Exception e) {
            System.out.println("Caught: " + e);
            System.out.println("Cause: " + e.getCause());
        }
    }

    public static void method1() throws Exception {
        try {
            method2();
        } catch (Exception e) {
            throw new Exception("Exception in method1", e);
        }
    }

    public static void method2() throws Exception {
        throw new Exception("Exception in method2");
    }
}
